    [resources 资源目录]

    [static 静态文件]
    [templates 模板文件] 
    [utils 全局工具] 
    [views 视图模块]
    [__init__.py create_app]
    [cli.py CMD配置]
    [extensions.py 扩展配置]
    [log.py 日志配置]
